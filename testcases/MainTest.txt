//画一条直线
//FOR T FROM 0 TO 1000 STEP 1 DRAW(T,T);
FOR T FROM 0 TO 1000 STEP 0.01 DRAW(T,EXP(LN(2*T-100)));

//设置原点为(300,300)
ORIGIN IS(300,300);

//画一条抛物线L1
SCALE IS(1,0.05);
FOR T FROM -1000 TO 1000 STEP 0.01 DRAW(T,0.2*(T-20)*(T-20));

//L1关于X轴对称画L2
SCALE IS(1,-0.05);
FOR T FROM -1000 TO 1000 STEP 0.01 DRAW(T,0.2*(T-20)*(T-20));

//L2旋转135°画L3
ROT IS 3*PI/4;
FOR T FROM -1000 TO 1000 STEP 0.01 DRAW(T,0.2*(T-20)*(T-20));
